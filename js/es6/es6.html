# 变量的解构赋值

1.数组的解构赋值

let [a, b, c] = [1, 2, 3]
// a = 1 b =2 c= 3


let [a, b, ...c] = [1, 2, 3,4 ,5, 6]
// a = 1 b =2 c= [3, 4, 5, 6]


let [a, [b], c] = [1, [2, 3], 4]
// a = 1 b = 2 c= 3


// 设置默认值
let [a, [b], c, d = 999] = [1, [2, 3], 4]
// 999

// 特殊的结构赋值
let arr = [1, 2, 3, 4]
let {0 : first, [arr.length - 1] : last} = arr;
// first 1
// last  4

2.对象的解构赋值

// 
let {a, b} = {a: 1, b: 2, c:3}
// a = 1  b= 2

// 起别名（重命名）
let {a: c, b: d} = {a: 1, b: 2, c:3}
// c = 1 d = 2

// 深层嵌套结构
obj = {
	a: {b: [1,2]}
}
// a 不会被结构出来
let {a: {b} } = obj
// b [1, 2]

// 结构复制的默认值
let {a, b, d = 999} = {a: 1, b: 2, c:3}
// d = 999

// 提前定义在赋值
let x;
({x} = {x: 1});

3.字符串的解构赋值(类数组)

// 装换为类数组结构赋值

    // 数组

    // 给默认值

        return x + y
    }
    fun()

    function fun({x, y}) {
        return x + y
    }
    fun({x: 1, y:2})

     function fun({x = 1, y = 2} = {}) {

    }
